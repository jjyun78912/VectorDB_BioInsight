class BaseAgent:
    """Abstract base class for all agents."""

    def __init__(self, name, input_dir, output_dir, config):
        self.name = name
        self.input_dir = Path(input_dir)
        self.output_dir = Path(output_dir)
        self.config = config
        self.logger = self._setup_logging()

    # Template Method Pattern
    def execute(self) -> Dict[str, Any]:
        """Execute the agent workflow."""
        self.logger.info(f"Starting {self.name}")

        # 1. Validate inputs
        if not self.validate_inputs():
            raise ValueError("Input validation failed")

        # 2. Run main logic
        results = self.run()

        # 3. Validate outputs
        if not self.validate_outputs():
            raise ValueError("Output validation failed")

        # 4. Save metadata
        self._save_metadata(results)

        return results

## Usage Examples

### Basic Usage

```python
from rnaseq_pipeline import RNAseqPipeline

# Initialize pipeline
pipeline = RNAseqPipeline(
    input_dir="./data/my_experiment",
    output_dir="./results",
    config={
        "cancer_type": "breast_cancer",
        "contrast": ["tumor", "normal"],
        "padj_cutoff": 0.05,
        "log2fc_cutoff": 1.0
    }
)

# Run full pipeline
results = pipeline.run()
print(f"Report: {results['run_dir']}/accumulated/report.html")
```

### CLI Usage

```bash
# Full pipeline
python -m rnaseq_pipeline.orchestrator \
    --input ./data/my_experiment \
    --output ./results \
    --cancer-type breast_cancer

# Resume from specific agent
python -m rnaseq_pipeline.orchestrator \
    --input ./data \
    --output ./results \
    --from-agent agent3_pathway

# Create sample data for testing
python -m rnaseq_pipeline.orchestrator \
    --input ./sample_data \
    --output ./results \
    --create-sample
```

### With ML Prediction

```python
from rnaseq_pipeline.ml import RNAseqPredictor

# Load pre-trained model
predictor = RNAseqPredictor.load("models/breast_cancer_classifier")

# Predict sample
result = predictor.predict_sample(expression_data)
print(f"Prediction: {result.prediction_label}")
print(f"Probability: {result.probability:.2%}")
print(f"Top genes: {result.top_genes}")
```
